# Usage

```javascript
import { QuanticProvider, createTheme, Button } from 'quantic-ui'

function App() {
  const theme = createTheme()

  return (
    <QuanticProvider theme={theme}>
      <Button onClick={() => console.log('clicked')}>Click Me</Button>
    </QuanticProvider>
  )
}

export default App
```
